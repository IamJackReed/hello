class TeamCheckIn {
    constructor(teamName) {
        this.teamName = teamName;
        this.responses = [];
    }
    
    addResponse(member, mood, concerns) {
        this.responses.push({
            member: member,
            mood: mood,
            concerns: concerns,
            timestamp: new Date()
        });
    }
}
